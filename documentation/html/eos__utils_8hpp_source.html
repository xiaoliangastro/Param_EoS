178 state_type cal_eborders(state_type params, state_type rho_borders){
179  state_type e_borders, gamma, pb, rb;
180  if ((param_method==PIECEWISE_PRESSURE) or (param_method==ADAPT_PIECEWISE) or (param_method==PIECE_SPEC_PHTR_CSS) or (param_method==PHYSICAL_SPEC_PHTR_CSS)){
181  for(int i=0; i<params.size(); i++){
182  if (i==0) gamma.push_back(log(params[0]/p_0)/log(rho_borders[0]/rho_0));
183  else gamma.push_back(log(params[i]/params[i-1])/log(rho_borders[i]/rho_borders[i-1]));
184  }
185  }
186  else if (param_method==PIECEWISE_GAMMA) {
187  for(int i=0; i<params.size(); i++) gamma.push_back(params[i]);
188  }
189  else { cout<<"Ask yourself: Why do you want to calculate eborders?!"<<endl; exit(1); }
190  int n_pieces = gamma.size();
191  double ai_plus1;
192  e_borders.push_back(e_0);
193  rb.push_back(rho_0);
194  pb.push_back(p_0);
195  for (int i=0; i<n_pieces; i++){
196  rb.push_back(rho_borders[i]);
197  pb.push_back(pb[i]*pow(rb[i+1]/rb[i],gamma[i]));
198  if (gamma[i]==1){
199  ai_plus1 = e_borders[i]/rb[i]-pb[i]*log(rb[i])/rb[i];
200  e_borders.push_back(rb[i+1]*ai_plus1+pb[i+1]*log(rb[i+1]));
201  }
202  else {
203  ai_plus1 = e_borders[i]/rb[i]-pb[i]/(rb[i]*(gamma[i]-1));
204  e_borders.push_back(rb[i+1]*ai_plus1+pb[i+1]/(gamma[i]-1));
205  }
206  }
207  if(verbose){
208  cout<<"gamma: ";
209  for(int i=0; i<gamma.size(); i++) cout<<gamma[i]<<" ";
210  cout<<endl;
211  cout<<"e_borders: ";
212  for(int i=0; i<e_borders.size(); i++) cout<<e_borders[i]<<" ";
213  cout<<endl;
214  }
215  return e_borders;
216 }

261  }
262  }
